179  print('%s: ' %name, end = '')

181  if self._to_skip > 0:

183  qutest._num_skipped += 1
184  print('SKIPPED')
185  self._tran(qutest._SKIP)
186  return
187 
188  if opt & qutest._OPT_NORESET != 0:
189  if self._state == qutest._FAIL:
190  self._fail('NORESET-test follows a failed test')
191  return

193  self._fail('NORESET-test needs reset')
194  return
195  self._tran(qutest._TEST)
196  else:
197  self._tran(qutest._TEST)

199  return

205  # expect DSL command .....................................................
206  def expect(self, match):
207  if self._to_skip > 0:
208  pass # ignore
209  elif self._state == qutest._INIT:
210  self._before_test('expect')
211  elif self._state == qutest._TEST:
212 
213  if not qspy._receive(): # timeout?
214  self._fail('got: "" (timeout)',
215  'exp: "%s"' %match)
216  return False
217 
218  if match.startswith('@timestamp'):
219  self._timestamp += 1
220  expected = '%010d' %self._timestamp + match[10:]
221  elif match[0:9].isdigit():
222  self._timestamp += 1
223  expected = match
224  else:
225  expected = match
226 
227  received = qutest._last_record[3:].decode('utf-8')
228 
229  if fnmatchcase(received, expected):
230  return True
231  else:
232  self._fail('got: "%s"' %received,
233  'exp: "%s"' %expected)
234  return False
235  elif self._state == qutest._FAIL or self._state == qutest._SKIP:
236  pass # ignore
237  else:
238  assert 0, 'invalid state in expect: ' + match

240  # glb_filter DSL command .................................................
241  def glb_filter(self, *args):
242  if self._to_skip > 0:
243  pass # ignore
244  elif self._state == qutest._INIT:
245  self._before_test('glb_filter')
246  elif self._state == qutest._TEST:
247  filter = [0, 0, 0, 0]
248 
249  for arg in args:
250  if arg == 0:
251  pass
252  elif arg < 0x7F:
253  filter[arg // 32] |= 1 << (arg % 32)
254  elif arg == qspy._GRP_ON:
255  # all filters on
256  filter[0] = 0xFFFFFFFF
257  filter[1] = 0xFFFFFFFF
258  filter[2] = 0xFFFFFFFF
259  filter[3] = 0x1FFFFFFF
260  break # no point in continuing
261  elif arg == qspy._GRP_SM: # state machines
262  filter[0] |= 0x000003FE
263  filter[1] |= 0x03800000
264  elif arg == qspy._GRP_AO: # active objects
265  filter[0] |= 0x0007FC00
266  filter[1] |= 0x00002000
267  elif arg == qspy._GRP_EQ: # raw queues
268  filter[0] |= 0x00780000
269  filter[2] |= 0x00004000
270  elif arg == qspy._GRP_MP: # raw memory pools
271  filter[0] |= 0x03000000
272  filter[2] |= 0x00008000
273  elif arg == qspy._GRP_QF: # framework
274  filter[0] |= 0xFC000000
275  filter[1] |= 0x00001FC0
276  elif arg == qspy._GRP_TE: # time events
277  filter[1] |= 0x0000007F
278  elif arg == qspy._GRP_SC: # scheduler
279  filter[1] |= 0x007F0000
280  elif arg == qspy._GRP_U0: # user 70-79
281  filter[2] |= 0x0000FFC0
282  elif arg == qspy._GRP_U1: # user 80-89
283  filter[2] |= 0x03FF0000
284  elif arg == qspy._GRP_U2: # user 90-99
285  filter[2] |= 0xFC000000
286  filter[3] |= 0x0000000F
287  elif arg == qspy._GRP_U3: # user 100-109
288  filter[3] |= 0x00003FF0
289  elif arg == qspy._GRP_U4: # user 110-124
290  filter[3] |= 0x1FFFC000
291  elif arg == qspy._GRP_UA: # user 70-124 (all)
292  filter[2] |= 0xFFFFFFC0
293  filter[3] |= 0x1FFFFFFF
294  else:
295  assert 0, 'invalid global filter'
296 
297  qspy._sendTo(struct.pack(
298  '<BBLLLL', qspy._TRGT_GLB_FILTER, 16,
299  filter[0], filter[1], filter[2], filter[3]))
300 
301  self.expect(' Trg-Ack QS_RX_GLB_FILTER')
302 
303  elif self._state == qutest._FAIL or self._state == qutest._SKIP:
304  pass # ignore
305  else:
306  assert 0, 'invalid state in glb_filter'
