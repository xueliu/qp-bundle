393 void QF_tickXISR(uint_fast8_t const tickRate) {
394  uint_fast8_t p = QF_maxActive_;
395  do {
396  QActive *a = QF_ROM_ACTIVE_GET_(p);
397  QTimer *t = &a->tickCtr[tickRate];
398 
399  if (t->nTicks != 0U) {
400  --t->nTicks;
401  if (t->nTicks == 0U) {
402 
403 #ifdef QF_TIMEEVT_PERIODIC
404  if (t->interval != 0U) {
405  t->nTicks = t->interval; /* re-arm the periodic timer */
406  }
407 #endif /* QF_TIMEEVT_PERIODIC */
408 
409 #ifdef QF_TIMEEVT_USAGE
410  QF_timerSetX_[tickRate] &= (uint_fast8_t)(~(1U << (p - 1U)));
411 #endif /* QF_TIMEEVT_USAGE */
412 
413 #if (Q_PARAM_SIZE != 0)
414  QACTIVE_POST_ISR(a, (enum_t)Q_TIMEOUT_SIG + (enum_t)tickRate,
415  0U);
416 #else
417  QACTIVE_POST_ISR(a, (enum_t)Q_TIMEOUT_SIG + (enum_t)tickRate);
418 #endif /* (Q_PARAM_SIZE != 0U) */
419  }
420  }
421  --p;
422  } while (p != 0U);
423 }
