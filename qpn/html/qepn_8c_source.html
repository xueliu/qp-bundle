103 void QHsm_init_(QHsm * const me) {
104  QStateHandler t = me->state;
105  QState r;
106 
111  Q_REQUIRE_ID(200, (me->vptr != (QHsmVtable const *)0)
112  && (me->temp != Q_STATE_CAST(0))
113  && (t == Q_STATE_CAST(&QHsm_top)));
114 
115  r = (*me->temp)(me); /* execute the top-most initial transition */
116 
117  /* the top-most initial transition must be taken */
118  Q_ASSERT_ID(210, r == Q_RET_TRAN);
119 
120  /* drill down into the state hierarchy with initial transitions... */
121  do {
122  QStateHandler path[QHSM_MAX_NEST_DEPTH_];
123  int_fast8_t ip = 0; /* transition entry path index */
124 
125  path[0] = me->temp;
126  Q_SIG(me) = QEP_EMPTY_SIG_;
127  (void)(*me->temp)(me);
128  while (me->temp != t) {
129  ++ip;
130  Q_ASSERT_ID(220, ip < (int_fast8_t)Q_DIM(path));
131  path[ip] = me->temp;
132  (void)(*me->temp)(me);
133  }
134  me->temp = path[0];
135 
136  /* retrace the entry path in reverse (desired) order... */
137  Q_SIG(me) = Q_ENTRY_SIG;
138  do {
139  (void)(*path[ip])(me); /* enter path[ip] */
140  --ip;
141  } while (ip >= 0);
142 
143  t = path[0]; /* current state becomes the new source */
144 
145  Q_SIG(me) = Q_INIT_SIG;
146  r = (*t)(me);
147  } while (r == Q_RET_TRAN);
148 
149  me->state = t; /* change the current active state */
150  me->temp = t; /* mark the configuration as stable */
151 }

186 void QHsm_dispatch_(QHsm * const me) {
187  QStateHandler t = me->state;
188  QStateHandler s;
189  QState r;
190  int_fast8_t iq; /* helper transition entry path index */
191 
195  Q_REQUIRE_ID(400, (t != Q_STATE_CAST(0))
196  && (t == me->temp));
197 
198  /* process the event hierarchically... */
199  do {
200  s = me->temp;
201  r = (*s)(me); /* invoke state handler s */
202 
203  if (r == Q_RET_UNHANDLED) { /* unhandled due to a guard? */
204  iq = (int_fast8_t)Q_SIG(me); /* save the original signal */
205  Q_SIG(me) = QEP_EMPTY_SIG_; /* find the superstate */
206  r = (*s)(me); /* invoke state handler s */
207  Q_SIG(me) = (QSignal)iq; /* restore the original signal */
208  }
209  } while (r == Q_RET_SUPER);
210 
211  /* transition taken? */
212  if (r >= Q_RET_TRAN) {
213  QStateHandler path[QHSM_MAX_NEST_DEPTH_]; /* transition entry path */
214  int_fast8_t ip; /* transition entry path index */
215 
216  path[0] = me->temp; /* save the target of the transition */
217  path[1] = t;
218  path[2] = s;
219 
220  /* exit current state to transition source s... */
221  for (; t != s; t = me->temp) {
222  Q_SIG(me) = Q_EXIT_SIG; /* find superstate of t */
223 
224  /* take the exit action and check if it was handled? */
225  if ((*t)(me) == Q_RET_HANDLED) {
226  Q_SIG(me) = QEP_EMPTY_SIG_;
227  (void)(*t)(me); /* find superstate of t */
228  }
229  }
230 
231  ip = QHsm_tran_(me, path); /* take the state transition */
232 
233  /* retrace the entry path in reverse (desired) order... */
234  Q_SIG(me) = Q_ENTRY_SIG;
235  for (; ip >= 0; --ip) {
236  (void)(*path[ip])(me); /* enter path[ip] */
237  }
238  t = path[0]; /* stick the target into register */
239  me->temp = t; /* update the current state */
240 
241  /* drill into the target hierarchy... */
242  Q_SIG(me) = Q_INIT_SIG;
243  while ((*t)(me) == Q_RET_TRAN) {
244  ip = 0;
245 
246  path[0] = me->temp;
247  Q_SIG(me) = QEP_EMPTY_SIG_;
248  (void)(*me->temp)(me); /* find the superstate */
249  while (me->temp != t) {
250  ++ip;
251  path[ip] = me->temp;
252  (void)(*me->temp)(me); /* find the superstate */
253  }
254  me->temp = path[0];
255 
256  /* entry path must not overflow */
257  Q_ASSERT_ID(410, ip < QHSM_MAX_NEST_DEPTH_);
258 
259  /* retrace the entry path in reverse (correct) order... */
260  Q_SIG(me) = Q_ENTRY_SIG;
261  do {
262  (void)(*path[ip])(me); /* enter path[ip] */
263  --ip;
264  } while (ip >= 0);
265 
266  t = path[0];
267  Q_SIG(me) = Q_INIT_SIG;
268  }
269  }
270 
271  me->state = t; /* change the current active state */
272  me->temp = t; /* mark the configuration as stable */
273 }

287 static int_fast8_t QHsm_tran_(QHsm * const me,
288  QStateHandler path[QHSM_MAX_NEST_DEPTH_])
289 {
290  int_fast8_t ip = (int_fast8_t)(-1); /* transition entry path index */
291  int_fast8_t iq; /* helper transition entry path index */
292  QStateHandler t = path[0];
293  QStateHandler s = path[2];
294  QState r;
295 
296  /* (a) check source==target (transition to self) */
297  if (s == t) {
298  Q_SIG(me) = Q_EXIT_SIG;
299  (void)(*s)(me); /* exit the source */
300  ip = 0; /* enter the target */
301  }
302  else {
303  Q_SIG(me) = QEP_EMPTY_SIG_;
304  (void)(*t)(me); /* find superstate of target */
305  t = me->temp;
306 
307  /* (b) check source==target->super */
308  if (s == t) {
309  ip = 0; /* enter the target */
310  }
311  else {
312  Q_SIG(me) = QEP_EMPTY_SIG_;
313  (void)(*s)(me); /* find superstate of source */
314 
315  /* (c) check source->super==target->super */
316  if (me->temp == t) {
317  Q_SIG(me) = Q_EXIT_SIG;
318  (void)(*s)(me); /* exit the source */
319  ip = 0; /* enter the target */
320  }
321  else {
322  /* (d) check source->super==target */
323  if (me->temp == path[0]) {
324  Q_SIG(me) = Q_EXIT_SIG;
325  (void)(*s)(me); /* exit the source */
326  }
327  else {
328  /* (e) check rest of source==target->super->super..
329  * and store the entry path along the way
330  */
331  iq = 0; /* indicate that LCA not found */
332  ip = 1; /* enter target and its superstate */
333  path[1] = t; /* save the superstate of target */
334  t = me->temp; /* save source->super */
335 
336  /* find target->super->super... */
337  Q_SIG(me) = QEP_EMPTY_SIG_;
338  r = (*path[1])(me);
339  while (r == Q_RET_SUPER) {
340  ++ip;
341  path[ip] = me->temp; /* store the entry path */
342  if (me->temp == s) { /* is it the source? */
343  iq = 1; /* indicate that LCA found */
344 
345  /* entry path must not overflow */
346  Q_ASSERT_ID(510, ip < QHSM_MAX_NEST_DEPTH_);
347  --ip; /* do not enter the source */
348  r = Q_RET_HANDLED; /* terminate loop */
349  }
350  /* it is not the source, keep going up */
351  else {
352  r = (*me->temp)(me); /* superstate of t */
353  }
354  }
355 
356  /* the LCA not found yet? */
357  if (iq == 0) {
358 
359  /* entry path must not overflow */
360  Q_ASSERT_ID(520, ip < QHSM_MAX_NEST_DEPTH_);
361 
362  Q_SIG(me) = Q_EXIT_SIG;
363  (void)(*s)(me); /* exit the source */
364 
365  /* (f) check the rest of source->super
366  * == target->super->super...
367  */
368  iq = ip;
369  r = Q_RET_IGNORED; /* LCA NOT found */
370  do {
371  if (t == path[iq]) { /* is this the LCA? */
372  r = Q_RET_HANDLED; /* LCA found */
373 
374  /* do not enter LCA */
375  ip = (int_fast8_t)(iq - 1);
376  iq = -1; /* force loop termination */
377  }
378  else {
379  --iq; /* try lower superstate of target */
380  }
381  } while (iq >= 0);
382 
383  /* LCA not found? */
384  if (r != Q_RET_HANDLED) {
385  /* (g) check each source->super->...
386  * for each target->super...
387  */
388  r = Q_RET_IGNORED; /* keep looping */
389  do {
390  /* exit t unhandled? */
391  Q_SIG(me) = Q_EXIT_SIG;
392  if ((*t)(me) == Q_RET_HANDLED) {
393  Q_SIG(me) = QEP_EMPTY_SIG_;
394  (void)(*t)(me); /* find super of t */
395  }
396  t = me->temp; /* set to super of t */
397  iq = ip;
398  do {
399  /* is this LCA? */
400  if (t == path[iq]) {
401  /* do not enter LCA */
402  ip = (int_fast8_t)(iq - 1);
403  iq = -1; /* break out of inner loop */
404  r = Q_RET_HANDLED; /* break */
405  }
406  else {
407  --iq;
408  }
409  } while (iq >= 0);
410  } while (r != Q_RET_HANDLED);
411  }
412  }
413  }
414  }
415  }
416  }
417  return ip;
418 }
