69 static void initialize(void) {
70  uint_fast8_t p;
71  QActive *a;
72 
73 #ifdef QF_MAX_ACTIVE /* deprecated constant provided? */
74 #if (QF_MAX_ACTIVE < 1) || (8 < QF_MAX_ACTIVE)
75  #error "QF_MAX_ACTIVE not defined or out of range. Valid range is 1..8"
76 #endif
77  QF_maxActive_ = (uint_fast8_t)QF_MAX_ACTIVE;
78 #else
79 
82  Q_REQUIRE_ID(100, (1U <= QF_maxActive_)
83  && (QF_maxActive_ <= (uint_fast8_t)8));
84 #endif
85 
86  /* set priorities all registered active objects... */
87  for (p = 1U; p <= QF_maxActive_; ++p) {
88  a = QF_ROM_ACTIVE_GET_(p);
89 
90  /* QF_active[p] must be initialized */
91  Q_ASSERT_ID(110, a != (QActive *)0);
92 
93  a->prio = (uint8_t)p; /* set the priority of the active object */
94  }
95 
96  /* trigger initial transitions in all registered active objects... */
97  for (p = 1U; p <= QF_maxActive_; ++p) {
98  a = QF_ROM_ACTIVE_GET_(p);
99  QHSM_INIT(&a->super); /* take the initial transition in the SM */
100  }
101 
102  /* process all events posted during initialization... */
103  QF_INT_DISABLE();
104  QK_attr_.actPrio = 0U; /* prio of the QK-nano idle loop */
105  if (QK_sched_() != 0U) {
106  QK_activate_(); /* activate AOs to process all events posted so far */
107  }
108  QF_INT_ENABLE();
109 }

161 QSchedStatus QK_schedLock(uint_fast8_t ceiling) {
162  QSchedStatus stat;
163  QF_INT_DISABLE();
164 
165  /* first store the previous lock prio */
166  /* raising the lock prio? */
167  if ((uint_fast8_t)QK_attr_.lockPrio < ceiling) {
168  stat = (QSchedStatus)QK_attr_.lockPrio << 8;
169  QK_attr_.lockPrio = (uint8_t)ceiling;
170 
171  /* add the previous lock holder priority */
172  stat |= (QSchedStatus)QK_attr_.lockHolder;
173 
174  QK_attr_.lockHolder = QK_attr_.actPrio;
175  }
176  else {
177  stat = 0xFFU;
178  }
179  QF_INT_ENABLE();
180 
181  return stat; /* return the status to be saved in a stack variable */
182 }

200 void QK_schedUnlock(QSchedStatus stat) {
201  /* has the scheduler been actually locked by the last QK_schedLock()? */
202  if (stat != 0xFFU) {
203  uint_fast8_t lockPrio = (uint_fast8_t)QK_attr_.lockPrio;
204  uint_fast8_t prevPrio = (uint_fast8_t)(stat >> 8);
205 
206  QF_INT_DISABLE();
207 
211  Q_REQUIRE_ID(700, lockPrio > prevPrio);
212 #ifdef Q_NASSERT
213  (void)lockPrio; /* avoid compiler warning about unused variable */
214 #endif
215 
216  /* restore the previous lock priority and lock holder */
217  QK_attr_.lockPrio = (uint8_t)prevPrio;
218  QK_attr_.lockHolder = (uint8_t)(stat & 0xFFU);
219 
220  /* find the highest-prio thread ready to run */
221  if (QK_sched_() != 0U) { /* priority found? */
222  QK_activate_(); /* activate any unlocked basic threads */
223  }
224 
225  QF_INT_ENABLE();
226  }
227 }

247 uint_fast8_t QK_sched_(void) {
248  uint_fast8_t p; /* for priority */
249 
250  /* find the highest-prio AO with non-empty event queue */
251 #ifdef QF_LOG2
252  p = (uint_fast8_t)QF_LOG2(QF_readySet_);
253 #else
254  /* hi nibble used? */
255  if ((QF_readySet_ & 0xF0U) != 0U) {
256  p = (uint_fast8_t)Q_ROM_BYTE(QF_log2Lkup[QF_readySet_ >> 4]) + 4U;
257  }
258  else { /* hi nibble of QF_readySet_ is zero */
259  p = (uint_fast8_t)Q_ROM_BYTE(QF_log2Lkup[QF_readySet_]);
260  }
261 #endif
262 
263  /* is the highest-prio below the active priority? */
264  if (p <= (uint_fast8_t)QK_attr_.actPrio) {
265  p = 0U; /* active object not eligible */
266  }
267 #ifdef QK_SCHED_LOCK
268  /* below the scheduler ceiling? */
269  else if (p <= (uint_fast8_t)QK_attr_.lockPrio) {
270  p = 0U; /* active object not eligible */
271  }
272 #endif /* QK_SCHED_LOCK */
273  else {
274  Q_ASSERT_ID(610, p <= QF_maxActive_);
275  QK_attr_.nextPrio = (uint8_t)p; /* next AO to run */
276  }
277 
278  return p;
279 }

291 void QK_activate_(void) {
292  uint_fast8_t pin = (uint_fast8_t)QK_attr_.actPrio; /* save active prio */
293  uint_fast8_t p = (uint_fast8_t)QK_attr_.nextPrio; /* next prio to run */
294 
295  /* QK Context switch callback enabled? */
296 #ifdef QK_ON_CONTEXT_SW
297  uint_fast8_t pprev = pin;
298 #endif /* QK_ON_CONTEXT_SW */
299 
300  /* QK_attr_.nextPrio must be non-zero upon entry to QK_activate_() */
301  Q_REQUIRE_ID(800, p != 0U);
302 
303  QK_attr_.nextPrio = 0U; /* clear for the next time */
304 
305  /* loop until no more ready-to-run AOs of higher prio than the initial */
306  do {
307  QActive *a;
308  QActiveCB const Q_ROM *acb;
309 
310  QK_attr_.actPrio = (uint8_t)p; /* this becomes the active priority */
311  QF_INT_ENABLE(); /* it's safe to leave critical section */
312 
313  acb = &QF_active[p];
314  a = QF_ROM_ACTIVE_GET_(p); /* map p to AO */
315 
316  QF_INT_DISABLE(); /* get ready to access the queue */
317 
318  /* some unused events must be available */
319  Q_ASSERT_ID(810, a->nUsed > 0U);
320  --a->nUsed;
321 
322  Q_SIG(a) = QF_ROM_QUEUE_AT_(acb, a->tail).sig;
323 #if (Q_PARAM_SIZE != 0)
324  Q_PAR(a) = QF_ROM_QUEUE_AT_(acb, a->tail).par;
325 #endif
326  /* wrap around? */
327  if (a->tail == 0U) {
328  a->tail = Q_ROM_BYTE(acb->qlen);
329  }
330  --a->tail;
331 
332 #ifdef QK_ON_CONTEXT_SW
333  if (p != pprev) { /* changing threads? */
334  QK_onContextSw(pprev, p); /* context-switch callback */
335  pprev = p; /* update previous priority */
336  }
337 #endif /* QK_ON_CONTEXT_SW */
338 
339  QF_INT_ENABLE(); /* unconditionally enable interrupts */
340 
341  QHSM_DISPATCH(&a->super); /* dispatch to the SM (execute RTC step) */
342 
343  QF_INT_DISABLE();
344 
345 
346  if (a->nUsed == 0U) { /* empty queue? */
347  /* clear the ready bit */
348  QF_readySet_ &= (uint_fast8_t)(~(1U << (p - 1U)));
349  }
350 
351  /* find new highest-prio AO ready to run... */
352 #ifdef QF_LOG2
353  p = (uint_fast8_t)QF_LOG2(QF_readySet_);
354 #else
355  /* hi nibble used? */
356  if ((QF_readySet_ & 0xF0U) != 0U) {
357  p = (uint_fast8_t)Q_ROM_BYTE(QF_log2Lkup[QF_readySet_ >> 4]) + 4U;
358  }
359  else { /* hi nibble of QF_readySet_ is zero */
360  p = (uint_fast8_t)Q_ROM_BYTE(QF_log2Lkup[QF_readySet_]);
361  }
362 #endif
363 
364  /* is the new priority below the initial preemption threshold? */
365  if (p <= pin) {
366  p = 0U; /* active object not eligible */
367  }
368 #ifdef QK_SCHED_LOCK
369  /* below the scheduler ceiling? */
370  else if (p <= (uint_fast8_t)QK_attr_.lockPrio) {
371  p = 0U; /* active object not eligible */
372  }
373  else {
374  Q_ASSERT_ID(710, p <= QF_maxActive_);
375  }
376 #endif /* QK_SCHED_LOCK */
377  } while (p != 0U);
378 
379  QK_attr_.actPrio = (uint8_t)pin; /* restore the active priority */
380 
381 #ifdef QK_ON_CONTEXT_SW
382  QK_onContextSw(pprev, pin); /* context-switch callback */
383 #endif /* QK_ON_CONTEXT_SW */
384 }
