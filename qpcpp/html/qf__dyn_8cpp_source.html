140 QEvt *QF::newX_(std::uint_fast16_t const evtSize,
141  std::uint_fast16_t const margin, enum_t const sig) noexcept
142 {
143  std::uint_fast8_t idx;
144 
145  // find the pool id that fits the requested event size ...
146  for (idx = 0U; idx < QF_maxPool_; ++idx) {
147  if (evtSize <= QF_EPOOL_EVENT_SIZE_(QF_pool_[idx])) {
148  break;
149  }
150  }
151  // cannot run out of registered pools
152  Q_ASSERT_ID(310, idx < QF_maxPool_);
153 
154  QS_CRIT_STAT_
155  QS_BEGIN_PRE_(QS_QF_NEW, nullptr, nullptr)
156  QS_TIME_PRE_(); // timestamp
157  QS_EVS_PRE_(evtSize); // the size of the evt
158  QS_SIG_PRE_(sig); // the signal of the evt
159  QS_END_PRE_()
160 
161  // get e -- platform-dependent
162  QEvt *e;
163  QF_EPOOL_GET_(QF_pool_[idx], e, ((margin != QF_NO_MARGIN) ? margin : 0U));
164 
165  // was e allocated correctly?
166  if (e != nullptr) {
167  e->sig = static_cast<QSignal>(sig); // set the signal
168  e->poolId_ = static_cast<std::uint8_t>(idx + 1U); // store pool ID
169  e->refCtr_ = 0U; // initialize the reference counter to 0
170  }
171  else {
172  // must tolerate bad alloc.
173  Q_ASSERT_ID(320, margin != QF_NO_MARGIN);
174  }
175  return e; // can't be NULL if we can't tolerate bad allocation
176 }

201 void QF::gc(QEvt const * const e) noexcept {
202  // is it a dynamic event?
203  if (e->poolId_ != 0U) {
204  QF_CRIT_STAT_
205  QF_CRIT_ENTRY_();
206 
207  // isn't this the last reference?
208  if (e->refCtr_ > 1U) {
209 
210  QS_BEGIN_NOCRIT_PRE_(QS_QF_GC_ATTEMPT, nullptr, nullptr)
211  QS_TIME_PRE_(); // timestamp
212  QS_SIG_PRE_(e->sig); // the signal of the event
213  QS_2U8_PRE_(e->poolId_, e->refCtr_); // pool Id & refCtr
214  QS_END_NOCRIT_PRE_()
215 
216  QF_EVT_REF_CTR_DEC_(e); // decrement the ref counter
217 
218  QF_CRIT_EXIT_();
219  }
220  // this is the last reference to this event, recycle it
221  else {
222  std::uint_fast8_t const idx =
223  static_cast<std::uint_fast8_t>(e->poolId_) - 1U;
224 
225  QS_BEGIN_NOCRIT_PRE_(QS_QF_GC, nullptr, nullptr)
226  QS_TIME_PRE_(); // timestamp
227  QS_SIG_PRE_(e->sig); // the signal of the event
228  QS_2U8_PRE_(e->poolId_, e->refCtr_);
229  QS_END_NOCRIT_PRE_()
230 
231  QF_CRIT_EXIT_();
232 
233  // pool ID must be in range
234  Q_ASSERT_ID(410, idx < QF_maxPool_);
235 
236 #ifdef Q_EVT_VIRTUAL
237  // explicitly exectute the destructor'
238  // NOTE: casting 'const' away is legitimate,
239  // because it's a pool event
240  QF_EVT_CONST_CAST_(e)->~QEvt(); // xtor,
241 #endif
242  // cast 'const' away, which is OK, because it's a pool event
243  QF_EPOOL_PUT_(QF_pool_[idx], QF_EVT_CONST_CAST_(e));
244  }
245  }
