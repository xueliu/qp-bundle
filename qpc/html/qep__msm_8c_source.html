187 void QMsm_dispatch_(QHsm * const me, QEvt const * const e) {
188  QMState const *s = me->state.obj; /* store the current state */
189  QMState const *t = s;
190  QState r;
191  QS_CRIT_STAT_
192 
194  Q_REQUIRE_ID(300, s != (QMState *)0);
195 
196  QS_BEGIN_PRE_(QS_QEP_DISPATCH, QS_priv_.locFilter[SM_OBJ], me)
197  QS_TIME_PRE_(); /* time stamp */
198  QS_SIG_PRE_(e->sig); /* the signal of the event */
199  QS_OBJ_PRE_(me); /* this state machine object */
200  QS_FUN_PRE_(s->stateHandler); /* the current state handler */
201  QS_END_PRE_()
202 
203  /* scan the state hierarchy up to the top state... */
204  do {
205  r = (*t->stateHandler)(me, e); /* call state handler function */
206 
207  /* event handled? (the most frequent case) */
208  if (r >= (QState)Q_RET_HANDLED) {
209  break; /* done scanning the state hierarchy */
210  }
211  /* event unhandled and passed to the superstate? */
212  else if (r == (QState)Q_RET_SUPER) {
213  t = t->superstate; /* advance to the superstate */
214  }
215  /* event unhandled and passed to a submachine superstate? */
216  else if (r == (QState)Q_RET_SUPER_SUB) {
217  t = me->temp.obj; /* current host state of the submachie */
218  }
219  /* event unhandled due to a guard? */
220  else if (r == (QState)Q_RET_UNHANDLED) {
221 
222  QS_BEGIN_PRE_(QS_QEP_UNHANDLED, QS_priv_.locFilter[SM_OBJ], me)
223  QS_SIG_PRE_(e->sig); /* the signal of the event */
224  QS_OBJ_PRE_(me); /* this state machine object */
225  QS_FUN_PRE_(t->stateHandler); /* the current state */
226  QS_END_PRE_()
227 
228  t = t->superstate; /* advance to the superstate */
229  }
230  else {
231  /* no other return value should be produced */
232  Q_ERROR_ID(310);
233  }
234  } while (t != (QMState *)0);
235 
236 
237  /* any kind of transition taken? */
238  if (r >= (QState)Q_RET_TRAN) {
239 #ifdef Q_SPY
240  QMState const *ts = t; /* transition source for QS tracing */
241 
242  /* the transition source state must not be NULL */
243  Q_ASSERT_ID(320, ts != (QMState *)0);
244 #endif /* Q_SPY */
245 
246  do {
247  /* save the transition-action table before it gets clobbered */
248  struct QMTranActTable const *tatbl = me->temp.tatbl;
249  union QHsmAttr tmp; /* temporary to save intermediate values */
250 
251  /* was TRAN, TRAN_INIT, or TRAN_EP taken? */
252  if (r <= (QState)Q_RET_TRAN_EP) {
253  QMsm_exitToTranSource_(me, s, t);
254  r = QMsm_execTatbl_(me, tatbl);
255  s = me->state.obj;
256  }
257  /* was a transition segment to history taken? */
258  else if (r == (QState)Q_RET_TRAN_HIST) {
259  tmp.obj = me->state.obj; /* save history */
260  me->state.obj = s; /* restore the original state */
261  QMsm_exitToTranSource_(me, s, t);
262  (void)QMsm_execTatbl_(me, tatbl);
263  r = QMsm_enterHistory_(me, tmp.obj);
264  s = me->state.obj;
265  }
266  /* was a transition segment to an exit point taken? */
267  else if (r == (QState)Q_RET_TRAN_XP) {
268  tmp.act = me->state.act; /* save XP action */
269  me->state.obj = s; /* restore the original state */
270  r = (*tmp.act)(me); /* execute the XP action */
271  if (r == (QState)Q_RET_TRAN) { /* XP -> TRAN ? */
272 #ifdef Q_SPY
273  tmp.tatbl = me->temp.tatbl; /* save me->temp */
274 #endif /* Q_SPY */
275  QMsm_exitToTranSource_(me, s, t);
276  /* take the tran-to-XP segment inside submachine */
277  (void)QMsm_execTatbl_(me, tatbl);
278  s = me->state.obj;
279 #ifdef Q_SPY
280  me->temp.tatbl = tmp.tatbl; /* restore me->temp */
281 #endif /* Q_SPY */
282  }
283  else if (r == (QState)Q_RET_TRAN_HIST) { /* XP -> HIST ? */
284  tmp.obj = me->state.obj; /* save the history */
285  me->state.obj = s; /* restore the original state */
286 #ifdef Q_SPY
287  s = me->temp.obj; /* save me->temp */
288 #endif /* Q_SPY */
289  QMsm_exitToTranSource_(me, me->state.obj, t);
290  /* take the tran-to-XP segment inside submachine */
291  (void)QMsm_execTatbl_(me, tatbl);
292 #ifdef Q_SPY
293  me->temp.obj = s; /* restore me->temp */
294 #endif /* Q_SPY */
295  s = me->state.obj;
296  me->state.obj = tmp.obj; /* restore the history */
297  }
298  else {
299  /* TRAN_XP must NOT be followed by any other tran type */
300  Q_ASSERT_ID(330, r < (QState)Q_RET_TRAN);
301  }
302  }
303  else {
304  /* no other return value should be produced */
305  Q_ERROR_ID(340);
306  }
307 
308  t = s; /* set target to the current state */
309 
310  } while (r >= (QState)Q_RET_TRAN);
311 
312  QS_BEGIN_PRE_(QS_QEP_TRAN, QS_priv_.locFilter[SM_OBJ], me)
313  QS_TIME_PRE_(); /* time stamp */
314  QS_SIG_PRE_(e->sig); /* the signal of the event */
315  QS_OBJ_PRE_(me); /* this state machine object */
316  QS_FUN_PRE_(ts->stateHandler); /* the transition source */
317  QS_FUN_PRE_(s->stateHandler); /* the new active state */
318  QS_END_PRE_()
319  }
320 
321 #ifdef Q_SPY
322  /* was the event handled? */
323  else if (r == (QState)Q_RET_HANDLED) {
324  /* internal tran. source can't be NULL */
325  Q_ASSERT_ID(340, t != (QMState *)0);
326 
327  QS_BEGIN_PRE_(QS_QEP_INTERN_TRAN, QS_priv_.locFilter[SM_OBJ], me)
328  QS_TIME_PRE_(); /* time stamp */
329  QS_SIG_PRE_(e->sig); /* the signal of the event */
330  QS_OBJ_PRE_(me); /* this state machine object */
331  QS_FUN_PRE_(t->stateHandler); /* the source state */
332  QS_END_PRE_()
333 
334  }
335  /* event bubbled to the 'top' state? */
336  else if (t == (QMState *)0) {
337 
338  QS_BEGIN_PRE_(QS_QEP_IGNORED, QS_priv_.locFilter[SM_OBJ], me)
339  QS_TIME_PRE_(); /* time stamp */
340  QS_SIG_PRE_(e->sig); /* the signal of the event */
341  QS_OBJ_PRE_(me); /* this state machine object */
342  QS_FUN_PRE_(s->stateHandler); /* the current state */
343  QS_END_PRE_()
344 
345  }
346 #endif /* Q_SPY */
347  else {
348  /* empty */
349  }
350 }

482 static QState QMsm_enterHistory_(QMsm * const me, QMState const *const hist) {
483  QMState const *s = hist;
484  QMState const *ts = me->state.obj; /* transition source */
485  QMState const *epath[QMSM_MAX_ENTRY_DEPTH_];
486  QState r;
487  uint_fast8_t i = 0U; /* transition entry path index */
488  QS_CRIT_STAT_
489 
490  QS_BEGIN_PRE_(QS_QEP_TRAN_HIST, QS_priv_.locFilter[SM_OBJ], me)
491  QS_OBJ_PRE_(me); /* this state machine object */
492  QS_FUN_PRE_(ts->stateHandler); /* source state handler */
493  QS_FUN_PRE_(hist->stateHandler); /* target state handler */
494  QS_END_PRE_()
495 
496  while (s != ts) {
497  if (s->entryAction != Q_ACTION_CAST(0)) {
498  epath[i] = s;
499  ++i;
500  Q_ASSERT_ID(620, i <= Q_DIM(epath));
501  }
502  s = s->superstate;
503  if (s == (QMState *)0) {
504  ts = s; /* force exit from the for-loop */
505  }
506  }
507 
508  /* retrace the entry path in reverse (desired) order... */
509  while (i > 0U) {
510  --i;
511  (void)(*epath[i]->entryAction)(me); /* run entry action in epath[i] */
512 
513  QS_BEGIN_PRE_(QS_QEP_STATE_ENTRY, QS_priv_.locFilter[SM_OBJ], me)
514  QS_OBJ_PRE_(me);
515  QS_FUN_PRE_(epath[i]->stateHandler); /* entered state handler */
516  QS_END_PRE_()
517  }
518 
519  me->state.obj = hist; /* set current state to the transition target */
520 
521  /* initial tran. present? */
522  if (hist->initAction != Q_ACTION_CAST(0)) {
523  r = (*hist->initAction)(me); /* execute the transition action */
524  }
525  else {
526  r = (QState)Q_RET_NULL;
527  }
528  return r;
529 }
