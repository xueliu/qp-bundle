413 static int_fast8_t QHsm_tran_(QHsm * const me,
414  QStateHandler path[QHSM_MAX_NEST_DEPTH_])
415 {
416  int_fast8_t ip = -1; /* transition entry path index */
417  int_fast8_t iq; /* helper transition entry path index */
418  QStateHandler t = path[0];
419  QStateHandler const s = path[2];
420  QState r;
421  QS_CRIT_STAT_
422 
423  /* (a) check source==target (transition to self)... */
424  if (s == t) {
425  QEP_EXIT_(s); /* exit the source */
426  ip = 0; /* enter the target */
427  }
428  else {
429  (void)QEP_TRIG_(t, QEP_EMPTY_SIG_); /* find superstate of target */
430 
431  t = me->temp.fun;
432 
433  /* (b) check source==target->super... */
434  if (s == t) {
435  ip = 0; /* enter the target */
436  }
437  else {
438  (void)QEP_TRIG_(s, QEP_EMPTY_SIG_); /* find superstate of src */
439 
440  /* (c) check source->super==target->super... */
441  if (me->temp.fun == t) {
442  QEP_EXIT_(s); /* exit the source */
443  ip = 0; /* enter the target */
444  }
445  else {
446  /* (d) check source->super==target... */
447  if (me->temp.fun == path[0]) {
448  QEP_EXIT_(s); /* exit the source */
449  }
450  else {
451  /* (e) check rest of source==target->super->super..
452  * and store the entry path along the way
453  */
454  iq = 0; /* indicate that LCA not found */
455  ip = 1; /* enter target and its superstate */
456  path[1] = t; /* save the superstate of target */
457  t = me->temp.fun; /* save source->super */
458 
459  /* find target->super->super... */
460  r = QEP_TRIG_(path[1], QEP_EMPTY_SIG_);
461  while (r == (QState)Q_RET_SUPER) {
462  ++ip;
463  path[ip] = me->temp.fun; /* store the entry path */
464  if (me->temp.fun == s) { /* is it the source? */
465  iq = 1; /* indicate that LCA found */
466 
467  /* entry path must not overflow */
468  Q_ASSERT_ID(510,
469  ip < QHSM_MAX_NEST_DEPTH_);
470  --ip; /* do not enter the source */
471  r = (QState)Q_RET_HANDLED; /* terminate loop */
472  }
473  /* it is not the source, keep going up */
474  else {
475  r = QEP_TRIG_(me->temp.fun, QEP_EMPTY_SIG_);
476  }
477  }
478 
479  /* the LCA not found yet? */
480  if (iq == 0) {
481 
482  /* entry path must not overflow */
483  Q_ASSERT_ID(520, ip < QHSM_MAX_NEST_DEPTH_);
484 
485  QEP_EXIT_(s); /* exit the source */
486 
487  /* (f) check the rest of source->super
488  * == target->super->super...
489  */
490  iq = ip;
491  r = (QState)Q_RET_IGNORED; /* LCA NOT found */
492  do {
493  if (t == path[iq]) { /* is this the LCA? */
494  r = (QState)Q_RET_HANDLED; /* LCA found */
495  ip = iq - 1; /* do not enter LCA */
496  iq = -1; /* cause termintion of the loop */
497  }
498  else {
499  --iq; /* try lower superstate of target */
500  }
501  } while (iq >= 0);
502 
503  /* LCA not found? */
504  if (r != (QState)Q_RET_HANDLED) {
505  /* (g) check each source->super->...
506  * for each target->super...
507  */
508  r = (QState)Q_RET_IGNORED; /* keep looping */
509  do {
510  /* exit t unhandled? */
511  if (QEP_TRIG_(t, Q_EXIT_SIG)
512  == (QState)Q_RET_HANDLED)
513  {
514  QS_BEGIN_PRE_(QS_QEP_STATE_EXIT,
515  QS_priv_.locFilter[SM_OBJ], me)
516  QS_OBJ_PRE_(me);
517  QS_FUN_PRE_(t);
518  QS_END_PRE_()
519 
520  (void)QEP_TRIG_(t, QEP_EMPTY_SIG_);
521  }
522  t = me->temp.fun; /* set to super of t */
523  iq = ip;
524  do {
525  /* is this LCA? */
526  if (t == path[iq]) {
527  /* do not enter LCA */
528  ip = (int_fast8_t)(iq - 1);
529  iq = -1; /* break out of inner loop */
530  /* break out of outer loop */
531  r = (QState)Q_RET_HANDLED;
532  }
533  else {
534  --iq;
535  }
536  } while (iq >= 0);
537  } while (r != (QState)Q_RET_HANDLED);
538  }
539  }
540  }
541  }
542  }
543  }
544  return ip;
