164 QEvt *QF_newX_(uint_fast16_t const evtSize,
165  uint_fast16_t const margin, enum_t const sig)
166 {
167  QEvt *e;
168  uint_fast8_t idx;
169  QS_CRIT_STAT_
170 
171  /* find the pool index that fits the requested event size ... */
172  for (idx = 0U; idx < QF_maxPool_; ++idx) {
173  if (evtSize <= QF_EPOOL_EVENT_SIZE_(QF_pool_[idx])) {
174  break;
175  }
176  }
177  /* cannot run out of registered pools */
178  Q_ASSERT_ID(310, idx < QF_maxPool_);
179 
180  QS_BEGIN_PRE_(QS_QF_NEW, (void *)0, (void *)0)
181  QS_TIME_PRE_(); /* timestamp */
182  QS_EVS_PRE_(evtSize); /* the size of the event */
183  QS_SIG_PRE_(sig); /* the signal of the event */
184  QS_END_PRE_()
185 
186  /* get e -- platform-dependent */
187  QF_EPOOL_GET_(QF_pool_[idx], e,
188  ((margin != QF_NO_MARGIN)
189  ? margin
190  : 0U));
191 
192  /* was e allocated correctly? */
193  if (e != (QEvt *)0) {
194  e->sig = (QSignal)sig; /* set signal for this event */
195  e->poolId_ = (uint8_t)(idx + 1U); /* store the pool ID */
196  e->refCtr_ = 0U; /* set the reference counter to 0 */
197  }
198  /* event cannot be allocated */
199  else {
200  /* must tolerate bad alloc. */
201  Q_ASSERT_ID(320, margin != QF_NO_MARGIN);
202  }
203  return e; /* can't be NULL if we can't tolerate bad allocation */
204 }

230 void QF_gc(QEvt const * const e) {
231 
232  /* is it a dynamic event? */
233  if (e->poolId_ != 0U) {
234  QF_CRIT_STAT_
235  QF_CRIT_ENTRY_();
236 
237  /* isn't this the last reference? */
238  if (e->refCtr_ > 1U) {
239 
240  QS_BEGIN_NOCRIT_PRE_(QS_QF_GC_ATTEMPT, (void *)0, (void *)0)
241  QS_TIME_PRE_(); /* timestamp */
242  QS_SIG_PRE_(e->sig); /* the signal of the event */
243  QS_2U8_PRE_(e->poolId_, e->refCtr_); /* pool Id & ref Count */
244  QS_END_NOCRIT_PRE_()
245 
246  QF_EVT_REF_CTR_DEC_(e); /* decrement the ref counter */
247 
248  QF_CRIT_EXIT_();
249  }
250  /* this is the last reference to this event, recycle it */
251  else {
252  uint_fast8_t idx = (uint_fast8_t)e->poolId_ - 1U;
253 
254  QS_BEGIN_NOCRIT_PRE_(QS_QF_GC, (void *)0, (void *)0)
255  QS_TIME_PRE_(); /* timestamp */
256  QS_SIG_PRE_(e->sig); /* the signal of the event */
257  QS_2U8_PRE_(e->poolId_, e->refCtr_); /* pool Id & ref Count */
258  QS_END_NOCRIT_PRE_()
259 
260  QF_CRIT_EXIT_();
261 
262  /* pool ID must be in range */
263  Q_ASSERT_ID(410, idx < QF_maxPool_);
264 
265  /* cast 'const' away, which is OK, because it's a pool event */
266  QF_EPOOL_PUT_(QF_pool_[idx], QF_EVT_CONST_CAST_(e));
267  }
268  }
