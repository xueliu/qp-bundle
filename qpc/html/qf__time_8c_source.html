76 #ifndef Q_SPY
77 void QF_tickX_(uint_fast8_t const tickRate)
78 #else
79 void QF_tickX_(uint_fast8_t const tickRate, void const * const sender)
80 #endif
81 {
82  QTimeEvt *prev = &QF_timeEvtHead_[tickRate];
83  QF_CRIT_STAT_
84 
85  QF_CRIT_ENTRY_();
86 
87  QS_BEGIN_NOCRIT_PRE_(QS_QF_TICK, (void *)0, (void *)0)
88  ++prev->ctr;
89  QS_TEC_PRE_(prev->ctr); /* tick ctr */
90  QS_U8_PRE_(tickRate); /* tick rate */
91  QS_END_NOCRIT_PRE_()
92 
93  /* scan the linked-list of time events at this rate... */
94  for (;;) {
95  QTimeEvt *t = prev->next; /* advance down the time evt. list */
96 
97  /* end of the list? */
98  if (t == (QTimeEvt *)0) {
99 
100  /* any new time events armed since the last run of QF_tickX_()? */
101  if (QF_timeEvtHead_[tickRate].act != (void *)0) {
102 
103  /* sanity check */
104  Q_ASSERT_CRIT_(110, prev != (QTimeEvt *)0);
105  prev->next = (QTimeEvt *)QF_timeEvtHead_[tickRate].act;
106  QF_timeEvtHead_[tickRate].act = (void *)0;
107  t = prev->next; /* switch to the new list */
108  }
109  else {
110  break; /* all currently armed time evts. processed */
111  }
112  }
113 
114  /* time event scheduled for removal? */
115  if (t->ctr == 0U) {
116  prev->next = t->next;
117  /* mark time event 't' as NOT linked */
118  t->super.refCtr_ &= (uint8_t)(~TE_IS_LINKED & 0xFFU);
119  /* do NOT advance the prev pointer */
120  QF_CRIT_EXIT_(); /* exit crit. section to reduce latency */
121 
122  /* prevent merging critical sections, see NOTE1 below */
123  QF_CRIT_EXIT_NOP();
124  }
125  else {
126  --t->ctr;
127 
128  /* is time event about to expire? */
129  if (t->ctr == 0U) {
130  QActive *act = (QActive *)t->act; /* temp. for volatile */
131 
132  /* periodic time evt? */
133  if (t->interval != 0U) {
134  t->ctr = t->interval; /* rearm the time event */
135  prev = t; /* advance to this time event */
136  }
137  /* one-shot time event: automatically disarm */
138  else {
139  prev->next = t->next;
140  /* mark time event 't' as NOT linked */
141  t->super.refCtr_ &= (uint8_t)(~TE_IS_LINKED & 0xFFU);
142  /* do NOT advance the prev pointer */
143 
144  QS_BEGIN_NOCRIT_PRE_(QS_QF_TIMEEVT_AUTO_DISARM,
145  QS_priv_.locFilter[TE_OBJ], t)
146  QS_OBJ_PRE_(t); /* this time event object */
147  QS_OBJ_PRE_(act); /* the target AO */
148  QS_U8_PRE_(tickRate); /* tick rate */
149  QS_END_NOCRIT_PRE_()
150  }
151 
152  QS_BEGIN_NOCRIT_PRE_(QS_QF_TIMEEVT_POST,
153  QS_priv_.locFilter[TE_OBJ], t)
154  QS_TIME_PRE_(); /* timestamp */
155  QS_OBJ_PRE_(t); /* the time event object */
156  QS_SIG_PRE_(t->super.sig); /* signal of this time event */
157  QS_OBJ_PRE_(act); /* the target AO */
158  QS_U8_PRE_(tickRate); /* tick rate */
159  QS_END_NOCRIT_PRE_()
160 
161  QF_CRIT_EXIT_(); /* exit critical section before posting */
162 
163  /* QACTIVE_POST() asserts internally if the queue overflows */
164  QACTIVE_POST(act, &t->super, sender);
165  }
166  else {
167  prev = t; /* advance to this time event */
168  QF_CRIT_EXIT_(); /* exit crit. section to reduce latency */
169 
170  /* prevent merging critical sections, see NOTE1 below */
171  QF_CRIT_EXIT_NOP();
172  }
173  }
174  QF_CRIT_ENTRY_(); /* re-enter crit. section to continue */
175  }
176  QF_CRIT_EXIT_();
177 }

179 /*****************************************************************************
180 * NOTE1:
181 * In some QF ports the critical section exit takes effect only on the next
182 * machine instruction. If this case, the next instruction is another entry
183 * to a critical section, the critical section won't be really exited, but
184 * rather the two adjacent critical sections would be merged.
185 *
186 * The QF_CRIT_EXIT_NOP() macro contains minimal code required
187 * to prevent such merging of critical sections in QF ports,
188 * in which it can occur.

239 void QTimeEvt_ctorX(QTimeEvt * const me, QActive * const act,
240  enum_t const sig, uint_fast8_t tickRate)
241 {
243  Q_REQUIRE_ID(300, (sig >= (enum_t)Q_USER_SIG)
244  && (tickRate < QF_MAX_TICK_RATE));
245 
246  me->next = (QTimeEvt *)0;
247  me->ctr = 0U;
248  me->interval = 0U;
249  me->super.sig = (QSignal)sig;
250 
251  /* For backwards compatibility with QTimeEvt_ctor(), the active object
252  * pointer can be uninitialized (NULL) and is NOT validated in the
253  * precondition. The active object pointer is validated in preconditions
254  * to QTimeEvt_arm_() and QTimeEvt_rearm().
255  */
256  me->act = act;
257 
258  /* Setting the POOL_ID event attribute to zero is correct only for
259  * events not allocated from event pools, which must be the case
260  * for Time Events.
261  */
262  me->super.poolId_ = 0U;
263 
264  /* The refCtr_ attribute is not used in time events, so it is
265  * reused to hold the tickRate as well as other information
266  */
267  me->super.refCtr_ = (uint8_t)tickRate;
268 }

298 void QTimeEvt_armX(QTimeEvt * const me,
299  QTimeEvtCtr const nTicks, QTimeEvtCtr const interval)
300 {
301  uint_fast8_t tickRate = ((uint_fast8_t)me->super.refCtr_ & TE_TICK_RATE);
302  QTimeEvtCtr ctr = me->ctr;
303  QF_CRIT_STAT_
304 
308  Q_REQUIRE_ID(400, (me->act != (void *)0)
309  && (ctr == 0U)
310  && (nTicks != 0U)
311  && (tickRate < (uint_fast8_t)QF_MAX_TICK_RATE)
312  && (me->super.sig >= (QSignal)Q_USER_SIG));
313 #ifdef Q_NASSERT
314  (void)ctr; /* avoid compiler warning about unused variable */
315 #endif
316 
317  QF_CRIT_ENTRY_();
318  me->ctr = nTicks;
319  me->interval = interval;
320 
321  /* is the time event unlinked?
322  * NOTE: For the duration of a single clock tick of the specified tick
323  * rate a time event can be disarmed and yet still linked into the list,
324  * because un-linking is performed exclusively in the QF_tickX() function.
325  */
326  if ((me->super.refCtr_ & TE_IS_LINKED) == 0U) {
327  me->super.refCtr_ |= TE_IS_LINKED; /* mark as linked */
328 
329  /* The time event is initially inserted into the separate
330  * "freshly armed" link list based on QF_timeEvtHead_[tickRate].act.
331  * Only later, inside the QF_tickX() function, the "freshly armed"
332  * list is appended to the main list of armed time events based on
333  * QF_timeEvtHead_[tickRate].next. Again, this is to keep any
334  * changes to the main list exclusively inside the QF_tickX()
335  * function.
336  */
337  me->next = (QTimeEvt *)QF_timeEvtHead_[tickRate].act;
338  QF_timeEvtHead_[tickRate].act = me;
339  }
340 
341  QS_BEGIN_NOCRIT_PRE_(QS_QF_TIMEEVT_ARM, QS_priv_.locFilter[TE_OBJ], me)
342  QS_TIME_PRE_(); /* timestamp */
343  QS_OBJ_PRE_(me); /* this time event object */
344  QS_OBJ_PRE_(me->act); /* the active object */
345  QS_TEC_PRE_(nTicks); /* the number of ticks */
346  QS_TEC_PRE_(interval); /* the interval */
347  QS_U8_PRE_(tickRate); /* tick rate */
348  QS_END_NOCRIT_PRE_()
349 
350  QF_CRIT_EXIT_();
351 }

374  QF_CRIT_STAT_

382

387  QS_OBJ_PRE_(me->act); /* the target AO */
388  QS_TEC_PRE_(me->ctr); /* the number of ticks */
389  QS_TEC_PRE_(me->interval); /* the interval */
390  QS_U8_PRE_(me->super.refCtr_ & TE_TICK_RATE);
391  QS_END_NOCRIT_PRE_()
392 
393  me->ctr = 0U; /* schedule removal from the list */
394  }
395  else { /* the time event was already disarmed automatically */

398

407  }

433 bool QTimeEvt_rearm(QTimeEvt * const me, QTimeEvtCtr const nTicks) {
434  uint_fast8_t tickRate = (uint_fast8_t)me->super.refCtr_ & TE_TICK_RATE;
435  bool wasArmed;
436  QF_CRIT_STAT_
437 
441  Q_REQUIRE_ID(600, (me->act != (void *)0)
442  && (tickRate < QF_MAX_TICK_RATE)
443  && (nTicks != 0U)
444  && (me->super.sig >= (QSignal)Q_USER_SIG));
445 
446  QF_CRIT_ENTRY_();
447 
448  /* is the time evt not running? */
449  if (me->ctr == 0U) {
450  wasArmed = false;
451 
452  /* NOTE: For the duration of a single clock tick of the specified
453  * tick rate a time event can be disarmed and yet still linked into
454  * the list, because unlinking is performed exclusively in the
455  * QF_tickX() function.
456  */
457  /* is the time event linked yet? */
458  if ((me->super.refCtr_ & TE_IS_LINKED) == 0U) {
459  me->super.refCtr_ |= TE_IS_LINKED; /* mark as linked */
460 
461  /* The time event is initially inserted into the separate
462  * "freshly armed" list based on QF_timeEvtHead_[tickRate].act.
463  * Only later, inside the QF_tickX() function, the "freshly armed"
464  * list is appended to the main list of armed time events based on
465  * QF_timeEvtHead_[tickRate].next. Again, this is to keep any
466  * changes to the main list exclusively inside the QF_tickX()
467  * function.
468  */
469  me->next = (QTimeEvt *)QF_timeEvtHead_[tickRate].act;
470  QF_timeEvtHead_[tickRate].act = me;
471  }
472  }
473  else { /* the time event was armed */
474  wasArmed = true;
475  }
476  me->ctr = nTicks; /* re-load the tick counter (shift the phasing) */
477 
478  QS_BEGIN_NOCRIT_PRE_(QS_QF_TIMEEVT_REARM, QS_priv_.locFilter[TE_OBJ], me)
479  QS_TIME_PRE_(); /* timestamp */
480  QS_OBJ_PRE_(me); /* this time event object */
481  QS_OBJ_PRE_(me->act); /* the target AO */
482  QS_TEC_PRE_(me->ctr); /* the number of ticks */
483  QS_TEC_PRE_(me->interval); /* the interval */
484  QS_2U8_PRE_(tickRate, (wasArmed ? 1U : 0U));
485  QS_END_NOCRIT_PRE_()
486 
487  QF_CRIT_EXIT_();
488  return wasArmed;
489 }
